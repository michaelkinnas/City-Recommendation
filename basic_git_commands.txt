git init 						#initialize git in working directory

git add . 						#add everything in the working directory to git

git status						#status of files

git remote add origin "<<REPOSITORY LINK FROM GITHUB>>"	#links git to github repository
git rm origin 						#removes remote repository
git remote -v 						#checks status of linked remote repository

git commit -a						#commit all changes to all files
git push origin main 					#push to master

git pull						#pull
git branch						#ελέγχω σε ποιο branch δουλεύω
